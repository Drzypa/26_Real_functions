\begin{eqnarray*}
\DDX\left[f(x)+g(x)\right]
& = & \lim_{h\to0}\frac{f(x+h)+g(x+h)-f(x)-g(x)}{h} \\
& = & \lim_{h\to0}\left[\frac{f(x+h)-f(x)}{h}+\frac{g(x+h)-g(x)}{h}\right] \\
& = & f'(x) + g'(x)
\end{eqnarray*}
